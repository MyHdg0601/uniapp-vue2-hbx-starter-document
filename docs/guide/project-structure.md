```
uniapp-vue2-hbx-starter
├─components                # 公共组件
├─enums                     # 枚举
├─extensions                # 扩展方法
├─filters                   # Vue过滤器
├─mixins                    # Vue混入
├─pages                     # 页面
├─service                   # 接口定义
├─static                    # 静态资源
├─store                     # VueX
├─styles                    # 样式表
├─utils                     # 配置/工具方法
├─App.vue                   # App.vue
├─index.html                # index.html
├─main.js                   # 应用入口
├─manifest.json             # uniapp配置
├─package.json              # package.json
├─pages.json                # 页面配置
├─theme.json                # 微信小程序主题配置
├─uni.scss                  # 全局scss变量
├─vue.config.js             # vue.config.js
├─.env                      # 通用环境变量
├─.env.development          # 开发环境变量
├─.env.production           # 生产环境变量
├─.settings                 # HBuilderX推荐配置
├─.editorconfig             # editorconfig
├─.gitignore                # git忽略文件
└-yarn.lock                 # 推荐使用Yarn管理依赖
```
